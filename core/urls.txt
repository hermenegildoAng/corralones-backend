from django.contrib import admin
from django.urls import path, include
from rest_framework.routers import DefaultRouter
from django.conf import settings
from django.urls import path
from django.conf.urls.static import static
from drf_spectacular.views import SpectacularAPIView, SpectacularSwaggerView
from rest_framework_simplejwt.views import TokenRefreshView
from registros import views  # ← ajusta 'registros' al nombre real de tu app Django
# Asegúrate de importar tus views correctamente
# QUITA crear_ingreso_completo de aquí:
from registros.views import (
    MyTokenObtainPairView,
    DepositoViewSet, UsuarioViewSet, VehiculoViewSet, 
    IngresoViewSet, PropietarioViewSet, DetallesAutoViewSet,
    ObjetoPersonalViewSet, RegistroDanoViewSet, InspeccionViewSet,
    SolicitudEdicionViewSet, BitacoraViewSet, Crear_solicitud_cambio
)

# 1. Definimos el router UNA SOLA VEZ
router = DefaultRouter()
router.register(r'depositos', DepositoViewSet)
router.register(r'usuarios', UsuarioViewSet)
router.register(r'vehiculos', VehiculoViewSet)
router.register(r'ingresos', IngresoViewSet, basename='ingresos')
router.register(r'propietarios', PropietarioViewSet)
router.register(r'detalles-auto', DetallesAutoViewSet)
router.register(r'objetos-personales', ObjetoPersonalViewSet)
router.register(r'danos', RegistroDanoViewSet)
router.register(r'inspecciones', InspeccionViewSet)
router.register(r'solicitudes-edicion', SolicitudEdicionViewSet)
router.register(r'crear-solicitud-cambio', Crear_solicitud_cambio)
router.register(r'bitacora', BitacoraViewSet)

# 2. Definimos los urlpatterns
urlpatterns = [
    path('admin/', admin.site.urls),
    
    # Ruta personalizada para el registro completo (debe ir antes del include del router)
    
    
    # Aquí se incluyen todas las rutas registradas arriba (usuarios, depositos, etc.)
    path('api/', include(router.urls)),
    
    # Autenticación
    path('api/login/', MyTokenObtainPairView.as_view(), name='token_obtain_pair'),
    path('api/token/refresh/', TokenRefreshView.as_view(), name='token_refresh'),
    
    # Documentación
    path('api/schema/', SpectacularAPIView.as_view(), name='schema'),
    path('docs/', SpectacularSwaggerView.as_view(url_name='schema'), name='swagger-ui'),

    path('api/password-reset/',         views.PasswordResetRequestView.as_view(), name='password_reset'),
    path('api/password-reset/confirm/', views.PasswordResetConfirmView.as_view(), name='password_reset_confirm'),
    path('ingresos/crear-solicitud-cambio/', views.Crear_solicitud_cambio, name='crear_solicitud_cambio'),
]

# Servir archivos media en desarrollo
if settings.DEBUG:
    urlpatterns += static(settings.MEDIA_URL, document_root=settings.MEDIA_ROOT)
